<div class="theorem">

1. $$
    \sin z = z \prod_{k=1}^{\infty}\cos\frac{z}{2^k}
   $$

2. $$
    \sin(\pi z) = \pi z \prod_{k=1}^{\infty}\left(1-\frac{z^2}{k^2}\right)
   $$

3. $$
    \cos(\pi z) = \prod_{k=1}^{\infty}\left(1-\frac{4z^2}{(2k-1)^2}\right)
   $$

4. $$
    \sinh z = z \prod_{k=1}^{\infty}\left(1+\frac{z^2}{k^2\pi^2}\right)
    $$

5. $$
    \cosh z = \prod_{k=1}^{\infty}\left(1+\frac{4z^2}{(2k-1)^2\pi^2}\right)
   $$

6. $$
    z! = \prod_{k=1}^{\infty}\frac{\left(1+\frac{1}{k}\right)^z}{1+\frac{z}{k}}

7. $$
    \sin(a+z) = (\sin a)\prod_{k=0,\pm1,\pm2,\ldots}^{\infty}\left(1+\frac{z}{a+k\pi}\right)
    $$

8. $$
    \cos(a+z) = (\cos a)\prod_{k=0,\pm1,\pm3,\pm5,\ldots}^{\infty}\left(1+\frac{2z}{2a+k\pi}\right)
   $$

</div>
